  INPUTS:
        z:             Zeeman_Input object to display.
                       UNITS:      N/A
                       TYPE:       Zeeman_Input_type
                       DIMENSION:  Scalar
                       ATTRIBUTES: INTENT(IN)
 
  \end{alltt}
