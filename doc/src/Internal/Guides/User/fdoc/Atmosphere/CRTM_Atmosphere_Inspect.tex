  NAME:

  INPUTS:
        Atm:  CRTM Atmosphere object to display.
              UNITS:      N/A
              TYPE:       CRTM_Atmosphere_type
              DIMENSION:  Scalar
              ATTRIBUTES: INTENT(IN)
 
  \end{alltt}
